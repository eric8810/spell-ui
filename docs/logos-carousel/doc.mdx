## Usage

```tsx
import { LogosCarousel } from "@/registry/spell-ui/logos-carousel";
```

```tsx
<LogosCarousel count={4}>
  <YourLogo1 />
  <YourLogo2 />
  <YourLogo3 />
  <YourLogo4 />
  <YourLogo5 />
  <YourLogo6 />
</LogosCarousel>
```
